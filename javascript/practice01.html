1.1 while문을 이용해서 100까지 출력하는 함수를 작성해주세요

{
    let i = 1;

    while (i <= 100) {
        console.log(i);
        i++;
    }
}

1.2 for문 이용해서 1부터 100까지 출력하는 함수를 작성해주세요.

{
    for(let i=1; i<=100; i++){
        console.log(i);
    }
}

조건1. 짝수만 출력하세요

조건2. 홀수만 출력하세요

{   
    //짝수
    for(let i=1; i<=100; i++){
        if(i % 2 == 0){
            console.log(i);
        }
    }

    //홀수
    for(let i=1; i<=100; i++){
        if(i % 2 !== 0){
            console.log(i);
        }
    }

    //다른방법
    for(let i=1; i<=100; i++){
        if(i % 2 == 0) console.log(i);
    }

        console.log(i);
    }  
}
